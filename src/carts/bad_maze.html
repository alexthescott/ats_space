function copy_table(t)
 n={}
	for k,v in pairs(t) do
	  n[k]=v
	end
	return n
end

function check_neigh(p, v, d)
	--given pos p, check
	--up right down left for unseen
	--up:0, right:1, down:2, left:3
	n={}

	--up
	if pget(p[1],p[2]+d) == v then
		add(n, {p[1],p[2]+d})
	end
	
	-- right
	if pget(p[1]+d,p[2]) == v then
			add(n, {p[1]+d,p[2]})
	end
	
	-- down
	if pget(p[1],p[2]-d) == v then
				add(n, {p[1],p[2]-d})
	end
	
	-- left
	if pget(p[1]-d,p[2]) == v then
				add(n, {p[1]-d,p[2]})
	end

	return n
end

function pop(stack)
    local v = stack[#stack]
    stack[#stack]=nil
    return v
end

	return n_pos
end
