alexthescott.single curved line

-- single curved line
-- alexthescott

o={}
inc=0.01
for i=0,2,inc do
	for c=0,2 do
		r=8+(i/2)*112
		a=(i/2)
		p=rnd()
		add(o,{r,a,p})
	end
end

cls()
_set_fps(60)
::♥::
for i=0,1024 do
	pset(rnd(128),rnd(128),0)
end

for i,ob in ipairs(o) do
	x=ob[1]*cos(ob[2])
	y=ob[1]*sin(ob[2])
	s=sin(ob[2])/2
	c=cos(ob[2])/2
	x=x*c-y*s+64
	y=x*s+y*c+64
	ob[2]+=inc/16
	ob[4]={x,y}
	if i>1 then
		line(o[i-1][4][1],o[i-1][4][2],x,y)
	end
	pset(x,y,7)
end

flip()
goto ♥
